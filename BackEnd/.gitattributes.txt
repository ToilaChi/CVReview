# Auto detect text files and normalize line endings to LF
* text=auto eol=lf

# Shell scripts must have LF (important for Docker containers)
*.sh text eol=lf

# Python files
*.py text eol=lf

# YAML files
*.yml text eol=lf
*.yaml text eol=lf

# Docker files
Dockerfile text eol=lf
docker-compose*.yml text eol=lf

# Java files
*.java text eol=lf
*.xml text eol=lf
*.properties text eol=lf

# Binary files
*.jar binary
*.class binary
*.so binary
*.pyc binary
